# Примеры запросов для проверки булевого поисковика (кино-тематика)
# Одна строка = один запрос. Комментарии начинаются с '#'.

# Простейшие терминные запросы
кино
режиссёр
актер

# Булевы комбинации (AND)
советский && режиссёр

# OR (альтернативы)
актер || актриса
режиссёр || сценарист

# NOT (исключение)
сериал && !мультфильм
комедия && !драма

# Короткие/общие термы
сериал
фильм && рецензия

# Многочастные запросы для стресс-тестов
актёр && режиссёр && продюсер

# Дополнительные случаи для проверки обработки дефисов/апострофов
рок-н-ролл || рок'n'roll
